import random
from words import words

print(random.choice(words))

def get_word():
    word=random.choice(words)
    while " " in word or "_" in word:
        word = random.choice(words)
    return word.upper()

# print(get_word())
def display(user_letters, word):
    display_word= ""
    for letter in word:
        if letter in user_letters:
            display_word += letter
        else:
            display_word += "*"
    return display_word
# print(display("as", "salom"))

def play():
    word=get_word()

    set_word = set(word)
    user_letters = ""

            continue
        if user_input in word:
            print("bu harf to'g'ri")
            set_word.remove(user_input)

        else:
            print("bunday harf mavjud emas")
        user_letters += user_input
    print(f"tabriklayman! siz {word} so'zini {len(user_letters)} ta" f"urinishda topdinggiz.")
